Лінії, прямокутники, еліпси, дуги, багатокутники

⏱️ Тривалість: 3 години 📖 Лекція 28 з 55

28.1 Фігури

Graphics^ g = e->Graphics;
Pen^ pen = gcnew Pen(Color::Cyan, 2);

// Прямокутник
Rectangle rect(10, 10, 150, 100);
g->DrawRectangle(pen, rect);
g->FillRectangle(Brushes::Blue, 170, 10, 150, 100);

// Еліпс
Rectangle ellipseRect(330, 10, 150, 100);
g->DrawEllipse(pen, ellipseRect);
g->FillEllipse(Brushes::Magenta, 10, 120, 150, 100);

// Дуга
g->DrawArc(pen, 170, 120, 150, 100, 0, 180);

// Багатокутник
array<Point>^ points = {
    Point(400, 120), Point(450, 150),
    Point(430, 200), Point(370, 200),
    Point(350, 150)
};
g->DrawPolygon(pen, points);
g->FillPolygon(gcnew SolidBrush(Color::Green), points);

// Крива Безьє
g->DrawBezier(pen, 10, 250, 100, 200, 200, 300, 300, 250);
