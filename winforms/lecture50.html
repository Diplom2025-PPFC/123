50.1 Проєкт: Змійка

public ref class SnakeForm : public Form {
private:
    Panel^ gamePanel;
    Timer^ gameTimer;
    List<Point>^ snake;
    Point food;
    Point direction;
    int cellSize = 20;
    int score = 0;
    bool gameRunning = false;
    Random^ rand;
    Label^ lblScore;

public:
    SnakeForm() {
        this->Text = "Змійка";
        this->Size = Size(640, 520);
        this->FormBorderStyle = FormBorderStyle::FixedSingle;
        this->BackColor = Color::FromArgb(10, 10, 20);

        // Панель гри
        gamePanel = gcnew Panel();
        gamePanel->Location = Point(10, 50);
        gamePanel->Size = Size(600, 400);
        gamePanel->BackColor = Color::FromArgb(20, 30, 20);
        gamePanel->BorderStyle = BorderStyle::FixedSingle;
        gamePanel->Paint += gcnew PaintEventHandler(
            this, &SnakeForm::OnGamePaint);
        this->Controls->Add(gamePanel);

        // Рахунок
        lblScore = gcnew Label();
        lblScore->Text = "Рахунок: 0";
        lblScore->Location = Point(10, 10);
        lblScore->ForeColor = Color::Lime;
        lblScore->Font = gcnew Font("Consolas", 16);
        this->Controls->Add(lblScore);

        // Таймер
        gameTimer = gcnew Timer();
        gameTimer->Interval = 100;
        gameTimer->Tick += gcnew EventHandler(
            this, &SnakeForm::OnGameTick);

        // Керування
        this->KeyDown += gcnew KeyEventHandler(
            this, &SnakeForm::OnKeyDown);

        rand = gcnew Random();
        InitGame();
    }

    void InitGame() {
        snake = gcnew List<Point>();
        snake->Add(Point(15, 10));
        snake->Add(Point(14, 10));
        snake->Add(Point(13, 10));
        direction = Point(1, 0);
        score = 0;
        SpawnFood();
        gameRunning = true;
        gameTimer->Start();
    }

    void SpawnFood() {
        food = Point(rand->Next(30), rand->Next(20));
    }

    void OnGameTick(Object^ s, EventArgs^ e) {
        // Нова голова
        Point head = snake[0];
        Point newHead = Point(head.X + direction.X, head.Y + direction.Y);

        // Перевірка зіткнення зі стіною
        if (newHead.X < 0 || newHead.X >= 30 || newHead.Y < 0 || newHead.Y >= 20) {
            GameOver();
            return;
        }

        // Перевірка зіткнення з собою
        for (int i = 0; i < snake->Count; i++) {
            if (snake[i] == newHead) {
                GameOver();
                return;
            }
        }

        snake->Insert(0, newHead);

        // Перевірка їжі
        if (newHead == food) {
            score += 10;
            lblScore->Text = "Рахунок: " + score;
            SpawnFood();
        } else {
            snake->RemoveAt(snake->Count - 1);
        }

        gamePanel->Invalidate();
    }

    void OnGamePaint(Object^ s, PaintEventArgs^ e) {
        Graphics^ g = e->Graphics;

        // Малювання змійки
        for (int i = 0; i < snake->Count; i++) {
            Color c = (i == 0) ? Color::Lime : Color::Green;
            g->FillRectangle(gcnew SolidBrush(c),
                snake[i].X * cellSize, snake[i].Y * cellSize,
                cellSize - 1, cellSize - 1);
        }

        // Малювання їжі
        g->FillRectangle(Brushes::Red,
            food.X * cellSize, food.Y * cellSize,
            cellSize - 1, cellSize - 1);
    }

    void OnKeyDown(Object^ s, KeyEventArgs^ e) {
        switch (e->KeyCode) {
            case Keys::Up: if (direction.Y == 0) direction = Point(0, -1); break;
            case Keys::Down: if (direction.Y == 0) direction = Point(0, 1); break;
            case Keys::Left: if (direction.X == 0) direction = Point(-1, 0); break;
            case Keys::Right: if (direction.X == 0) direction = Point(1, 0); break;
        }
    }

    void GameOver() {
        gameTimer->Stop();
        gameRunning = false;
        MessageBox::Show("Гру закінчено! Рахунок: " + score);
        InitGame();
    }
};
