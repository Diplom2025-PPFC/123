49.1 Проєкт: Хрестики-нулики

public ref class TicTacToeForm : public Form {
private:
    array<Button^, 2>^ buttons;
    array<array<int>^>^ board;
    int currentPlayer = 1; // 1 = X, 2 = O
    int movesCount = 0;
    Label^ lblStatus;
    bool gameOver = false;

public:
    TicTacToeForm() {
        this->Text = "Хрестики-нулики";
        this->Size = Size(400, 500);
        this->BackColor = Color::FromArgb(20, 20, 40);

        buttons = gcnew array<Button^, 2>(3, 3);
        board = gcnew array<array<int>^>(3);
        for (int i = 0; i < 3; i++) board[i] = gcnew array<int>(3);

        // Ігрове поле
        for (int row = 0; row < 3; row++) {
            for (int col = 0; col < 3; col++) {
                Button^ btn = gcnew Button();
                btn->Size = Size(100, 100);
                btn->Location = Point(20 + col * 105, 20 + row * 105);
                btn->Font = gcnew Font("Arial", 36, FontStyle::Bold);
                btn->Tag = gcnew array<int>{row, col};
                btn->FlatStyle = FlatStyle::Flat;
                btn->FlatAppearance->BorderSize = 2;
                btn->FlatAppearance->BorderColor = Color::Cyan;
                btn->BackColor = Color::FromArgb(40, 40, 60);
                btn->ForeColor = Color::White;
                btn->Click += gcnew EventHandler(
                    this, &TicTacToeForm::OnCellClick);
                buttons[row, col] = btn;
                this->Controls->Add(btn);
            }
        }

        // Статус
        lblStatus = gcnew Label();
        lblStatus->Text = "Хід гравця X";
        lblStatus->Location = Point(20, 340);
        lblStatus->Size = Size(300, 30);
        lblStatus->ForeColor = Color::Cyan;
        lblStatus->Font = gcnew Font("Segoe UI", 14);
        this->Controls->Add(lblStatus);

        // Кнопка нової гри
        Button^ btnNew = gcnew Button();
        btnNew->Text = "Нова гра";
        btnNew->Location = Point(20, 380);
        btnNew->Size = Size(100, 40);
        btnNew->Click += gcnew EventHandler(
            this, &TicTacToeForm::OnNewGame);
        this->Controls->Add(btnNew);
    }

    void OnCellClick(Object^ s, EventArgs^ e) {
        if (gameOver) return;

        Button^ btn = (Button^)s;
        array<int>^ pos = (array<int>^)btn->Tag;
        int row = pos[0], col = pos[1];

        if (board[row][col] != 0) return;

        board[row][col] = currentPlayer;
        btn->Text = (currentPlayer == 1) ? "X" : "O";
        btn->ForeColor = (currentPlayer == 1) ? Color::Cyan : Color::Magenta;
        movesCount++;

        if (CheckWinner(row, col)) {
            lblStatus->Text = String::Format("Гравець {0} переміг!",
                (currentPlayer == 1) ? "X" : "O");
            gameOver = true;
            return;
        }

        if (movesCount == 9) {
            lblStatus->Text = "Нічия!";
            gameOver = true;
            return;
        }

        currentPlayer = (currentPlayer == 1) ? 2 : 1;
        lblStatus->Text = String::Format("Хід гравця {0}",
            (currentPlayer == 1) ? "X" : "O");
    }

    bool CheckWinner(int row, int col) {
        int p = board[row][col];
        // Рядок
        if (board[row][0] == p && board[row][1] == p && board[row][2] == p) return true;
        // Стовпець
        if (board[0][col] == p && board[1][col] == p && board[2][col] == p) return true;
        // Діагоналі
        if (board[0][0] == p && board[1][1] == p && board[2][2] == p) return true;
        if (board[0][2] == p && board[1][1] == p && board[2][0] == p) return true;
        return false;
    }

    void OnNewGame(Object^ s, EventArgs^ e) {
        for (int r = 0; r < 3; r++)
            for (int c = 0; c < 3; c++) {
                board[r][c] = 0;
                buttons[r, c]->Text = "";
            }
        currentPlayer = 1;
        movesCount = 0;
        gameOver = false;
        lblStatus->Text = "Хід гравця X";
    }
};
