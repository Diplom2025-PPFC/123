13.1 ListView

// Таблиця
ListView^ lv = gcnew ListView();
lv->Location = Point(20, 20);
lv->Size = Size(500, 300);
lv->View = View::Details;
lv->FullRowSelect = true;
lv->GridLines = true;

// Колонки
lv->Columns->Add("Ім'я", 150);
lv->Columns->Add("Розмір", 100);
lv->Columns->Add("Тип", 100);
lv->Columns->Add("Дата", 150);

// Рядки
ListViewItem^ item1 = gcnew ListViewItem(gcnew array<String^> {
    "document.txt", "15 KB", "Текстовий файл", "01.01.2024"
});
ListViewItem^ item2 = gcnew ListViewItem(gcnew array<String^> {
    "photo.jpg", "2.5 MB", "Зображення", "05.01.2024"
});
lv->Items->Add(item1);
lv->Items->Add(item2);
