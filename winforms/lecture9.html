Лекція 9: Числовий вибір NumericUpDown

Вибір числових значень, крок, мін/макс, десяткові

⏱️ Тривалість: 3 години 📖 Лекція 9 з 55

9.1 NumericUpDown

// Числовий вибір
NumericUpDown^ nud = gcnew NumericUpDown();
nud->Location = Point(20, 20);
nud->Size = Size(120, 25);
nud->Minimum = 0;
nud->Maximum = 100;
nud->Value = 50;
nud->Increment = 5;
nud->DecimalPlaces = 0;

// Десяткове значення
NumericUpDown^ nudPrice = gcnew NumericUpDown();
nudPrice->Minimum = 0;
nudPrice->Maximum = 9999.99;
nudPrice->DecimalPlaces = 2;
nudPrice->Increment = 0.5;

// Подія ValueChanged
nud->ValueChanged += gcnew EventHandler(
    this, &MyForm::OnValueChanged);

void OnValueChanged(Object^ sender, EventArgs^ e) {
    NumericUpDown^ n = (NumericUpDown^)sender;
    lblResult->Text = "Значення: " + n->Value.ToString();
}
