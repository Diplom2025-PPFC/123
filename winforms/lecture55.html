55.1 Проєкт: Система управління бібліотекою

// Основна форма з TabControl
public ref class LibraryForm : public Form {
private:
    TabControl^ tabs;
    SqlConnection^ conn;

    // Вкладка Книги
    DataGridView^ dgvBooks;
    TextBox^ txtBookTitle, ^txtAuthor, ^txtYear, ^txtISBN;

    // Вкладка Читачі
    DataGridView^ dgvReaders;
    TextBox^ txtReaderName, ^txtReaderPhone;

    // Вкладка Видача
    DataGridView^ dgvLoans;
    ComboBox^ cmbBook, ^cmbReader;

public:
    LibraryForm() {
        this->Text = "Система управління бібліотекою";
        this->Size = Size(1000, 700);

        String^ connStr = "Server=(localdb)\MSSQLLocalDB;"
            "Database=LibraryDB;Integrated Security=true;";
        conn = gcnew SqlConnection(connStr);

        tabs = gcnew TabControl();
        tabs->Dock = DockStyle::Fill;

        // Вкладка Книги
        TabPage^ tabBooks = gcnew TabPage("Книги");
        SetupBooksTab(tabBooks);
        tabs->TabPages->Add(tabBooks);

        // Вкладка Читачі
        TabPage^ tabReaders = gcnew TabPage("Читачі");
        SetupReadersTab(tabReaders);
        tabs->TabPages->Add(tabReaders);

        // Вкладка Видача
        TabPage^ tabLoans = gcnew TabPage("Видача/Повернення");
        SetupLoansTab(tabLoans);
        tabs->TabPages->Add(tabLoans);

        // Вкладка Звіти
        TabPage^ tabReports = gcnew TabPage("Звіти");
        SetupReportsTab(tabReports);
        tabs->TabPages->Add(tabReports);

        this->Controls->Add(tabs);
    }

    void SetupBooksTab(TabPage^ tab) {
        // Таблиця книг
        dgvBooks = gcnew DataGridView();
        dgvBooks->Location = Point(10, 10);
        dgvBooks->Size = Size(700, 300);
        dgvBooks->AutoSizeColumnsMode = DataGridViewAutoSizeColumnsMode::Fill;
        dgvBooks->SelectionChanged += gcnew EventHandler(
            this, &LibraryForm::OnBookSelect);
        tab->Controls->Add(dgvBooks);

        // Поля
        int y = 330;
        AddBookField(tab, "Назва:", txtBookTitle = gcnew TextBox(), y);
        AddBookField(tab, "Автор:", txtAuthor = gcnew TextBox(), y += 30);
        AddBookField(tab, "Рік:", txtYear = gcnew TextBox(), y += 30);
        AddBookField(tab, "ISBN:", txtISBN = gcnew TextBox(), y += 30);

        // Кнопки
        Button^ btnAdd = gcnew Button();
        btnAdd->Text = "Додати книгу";
        btnAdd->Location = Point(10, y + 40);
        btnAdd->Click += gcnew EventHandler(
            this, &LibraryForm::OnAddBook);
        tab->Controls->Add(btnAdd);

        Button^ btnSearch = gcnew Button();
        btnSearch->Text = "Пошук";
        btnSearch->Location = Point(120, y + 40);
        btnSearch->Click += gcnew EventHandler(
            this, &LibraryForm::OnSearchBook);
        tab->Controls->Add(btnSearch);

        LoadBooks();
    }

    void AddBookField(TabPage^ tab, String^ label, TextBox^ txt, int y) {
        Label^ lbl = gcnew Label();
        lbl->Text = label;
        lbl->Location = Point(10, y);
        lbl->AutoSize = true;
        tab->Controls->Add(lbl);

        txt->Location = Point(120, y);
        txt->Size = Size(200, 25);
        tab->Controls->Add(txt);
    }

    void LoadBooks() {
        SqlDataAdapter^ adapter = gcnew SqlDataAdapter(
            "SELECT * FROM Books", conn);
        DataTable^ table = gcnew DataTable();
        adapter->Fill(table);
        dgvBooks->DataSource = table;
    }

    void OnAddBook(Object^ s, EventArgs^ e) {
        SqlCommand^ cmd = gcnew SqlCommand(
            "INSERT INTO Books (Title, Author, Year, ISBN, Available) "
            "VALUES (@title, @author, @year, @isbn, 1)", conn);
        cmd->Parameters->AddWithValue("@title", txtBookTitle->Text);
        cmd->Parameters->AddWithValue("@author", txtAuthor->Text);
        cmd->Parameters->AddWithValue("@year", Convert::ToInt32(txtYear->Text));
        cmd->Parameters->AddWithValue("@isbn", txtISBN->Text);

        conn->Open();
        cmd->ExecuteNonQuery();
        conn->Close();
        LoadBooks();
    }

    void OnBookSelect(Object^ s, EventArgs^ e) {
        if (dgvBooks->SelectedRows->Count > 0) {
            txtBookTitle->Text = dgvBooks->SelectedRows[0]->Cells["Title"]->Value->ToString();
            txtAuthor->Text = dgvBooks->SelectedRows[0]->Cells["Author"]->Value->ToString();
        }
    }

    void OnSearchBook(Object^ s, EventArgs^ e) {
        SqlDataAdapter^ adapter = gcnew SqlDataAdapter(
            "SELECT * FROM Books WHERE Title LIKE @search OR Author LIKE @search",
            conn);
        adapter->SelectCommand->Parameters->AddWithValue(
            "@search", "%" + txtBookTitle->Text + "%");
        DataTable^ table = gcnew DataTable();
        adapter->Fill(table);
        dgvBooks->DataSource = table;
    }

    void SetupReadersTab(TabPage^ tab) { /* Аналогічно */ }
    void SetupLoansTab(TabPage^ tab) { /* Аналогічно */ }
    void SetupReportsTab(TabPage^ tab) { /* Chart, ReportViewer */ }
};

55.2 Структура бази даних

-- SQL для створення бази бібліотеки
CREATE DATABASE LibraryDB;
GO
USE LibraryDB;
GO

CREATE TABLE Books (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Title NVARCHAR(200) NOT NULL,
    Author NVARCHAR(100),
    Year INT,
    ISBN NVARCHAR(20) UNIQUE,
    Available BIT DEFAULT 1
);

CREATE TABLE Readers (
    Id INT PRIMARY KEY IDENTITY(1,1),
    Name NVARCHAR(100) NOT NULL,
    Phone NVARCHAR(20),
    Email NVARCHAR(100),
    Address NVARCHAR(200)
);

CREATE TABLE Loans (
    Id INT PRIMARY KEY IDENTITY(1,1),
    BookId INT FOREIGN KEY REFERENCES Books(Id),
    ReaderId INT FOREIGN KEY REFERENCES Readers(Id),
    LoanDate DATE DEFAULT GETDATE(),
    ReturnDate DATE,
    Status NVARCHAR(20) DEFAULT 'Active'
);
