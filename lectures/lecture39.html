Лекція 39: Перевантаження операторів

Оператори як методи класу, friend оператори, правила

39.1 Перевантаження операторів

C++ дозволяє визначати поведінку операторів для користувацьких типів.

#include <iostream>

class Complex {
private:
    double real;
    double imag;

public:
    Complex(double r = 0, double i = 0) : real(r), imag(i) {}

    // Оператор + як метод класу
    Complex operator+(const Complex& other) const {
        return Complex(real + other.real, imag + other.imag);
    }

    // Оператор - як метод класу
    Complex operator-(const Complex& other) const {
        return Complex(real - other.real, imag - other.imag);
    }

    // Оператор * як метод класу
    Complex operator*(const Complex& other) const {
        return Complex(
            real * other.real - imag * other.imag,
            real * other.imag + imag * other.real
        );
    }

    // Оператор ==
    bool operator==(const Complex& other) const {
        return (real == other.real && imag == other.imag);
    }

    // Оператор виведення (friend)
    friend std::ostream& operator<<(std::ostream& os, const Complex& c) {
        os << c.real;
        if (c.imag >= 0) os << "+";
        os << c.imag << "i";
        return os;
    }

    // Оператор введення (friend)
    friend std::istream& operator>>(std::istream& is, Complex& c) {
        std::cout << "Дійсна частина: ";
        is >> c.real;
        std::cout << "Уявна частина: ";
        is >> c.imag;
        return is;
    }
};

int main() {
    Complex a(3, 4);
    Complex b(1, 2);

    Complex c = a + b;
    Complex d = a - b;
    Complex e = a * b;

    std::cout << "a = " << a << std::endl;
    std::cout << "b = " << b << std::endl;
    std::cout << "a + b = " << c << std::endl;
    std::cout << "a - b = " << d << std::endl;
    std::cout << "a * b = " << e << std::endl;

    return 0;
}

39.2 Підсумки лекції

  • Оператори можна перевантажувати для класів
  • Деякі оператори не можна перевантажувати
  • friend для операторів введення/виведення
