Лекція 31: Інкапсуляція: модифікатори доступу

private, public, protected, доступ до членів класу

31.1 Модифікатори доступу

Інкапсуляція приховує внутрішню реалізацію класу від зовнішнього світу. Доступ до даних контролюється через модифікатори:

#include <iostream>
#include <string>

class BankAccount {
private:    // Доступно тільки всередині класу
    std::string owner;
    double balance;
    std::string password;

protected:  // Доступно в класі та похідних класах
    int accountType;

public:     // Доступно звідусіль
    // Конструктор
    BankAccount(std::string name, double initial, std::string pwd) 
        : owner(name), balance(initial), password(pwd), accountType(1) {}

    // Публічний інтерфейс
    void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            std::cout << "Зараховано: " << amount << std::endl;
        }
    }

    bool withdraw(double amount, std::string pwd) {
        if (pwd != password) {
            std::cout << "Невірний пароль!" << std::endl;
            return false;
        }
        if (amount > balance) {
            std::cout << "Недостатньо коштів!" << std::endl;
            return false;
        }
        balance -= amount;
        std::cout << "Знято: " << amount << std::endl;
        return true;
    }

    double getBalance() const {
        return balance;
    }
};

int main() {
    BankAccount account("Ivan Petrov", 1000.0, "secret123");

    account.deposit(500);
    account.withdraw(200, "secret123");
    std::cout << "Баланс: " << account.getBalance() << std::endl;

    // account.balance = 999999;  // ПОМИЛКА! balance — private

    return 0;
}

31.2 Підсумки лекції

  • private — тільки всередині класу
  • protected — в класі та похідних
  • public — доступно всюди
  • Інкапсуляція захищає дані
