Лекція 13: Вкладені цикли

13.1 Концепція вкладених циклів

Вкладений цикл — це цикл всередині іншого циклу. Зовнішній цикл контролює рядки/ітерації верхнього рівня, внутрішній — стовпці/деталі кожної ітерації.

// Синтаксис вкладених циклів:
for (int i = 0; i < rows; i++) {      // зовнішній цикл
    for (int j = 0; j < cols; j++) {  // внутрішній цикл
        // дії
    }
}

// Приклад: матриця чисел
#include <iostream>
int main() {
    for (int i = 1; i <= 3; i++) {
        for (int j = 1; j <= 3; j++) {
            std::cout << i * j << "\t";
        }
        std::cout << std::endl;
    }
    // Виведе:
    // 1  2  3
    // 2  4  6
    // 3  6  9
    return 0;
}

13.2 Малювання фігур

// Квадрат з зірочок
#include <iostream>
int main() {
    int size = 5;
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            std::cout << "* ";
        }
        std::cout << std::endl;
    }
    // * * * * *
    // * * * * *
    // * * * * *
    // * * * * *
    // * * * * *
    return 0;
}

// Прямокутний трикутник
#include <iostream>
int main() {
    int height = 5;
    for (int i = 1; i <= height; i++) {
        for (int j = 1; j <= i; j++) {
            std::cout << "* ";
        }
        std::cout << std::endl;
    }
    // *
    // * *
    // * * *
    // * * * *
    // * * * * *
    return 0;
}

// Піраміда
#include <iostream>
int main() {
    int height = 5;
    for (int i = 1; i <= height; i++) {
        // Пробіли
        for (int j = 1; j <= height - i; j++) {
            std::cout << " ";
        }
        // Зірочки
        for (int k = 1; k <= 2 * i - 1; k++) {
            std::cout << "*";
        }
        std::cout << std::endl;
    }
    //     *
    //    ***
    //   *****
    //  *******
    // *********
    return 0;
}

// Ромб
#include <iostream>
int main() {
    int n = 5;
    // Верхня частина
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n - i; j++) std::cout << " ";
        for (int k = 1; k <= 2 * i - 1; k++) std::cout << "*";
        std::cout << std::endl;
    }
    // Нижня частина
    for (int i = n - 1; i >= 1; i--) {
        for (int j = 1; j <= n - i; j++) std::cout << " ";
        for (int k = 1; k <= 2 * i - 1; k++) std::cout << "*";
        std::cout << std::endl;
    }
    return 0;
}

13.3 Таблиця множення

#include <iostream>
#include <iomanip>

int main() {
    std::cout << "    Таблиця множення" << std::endl;
    std::cout << "    ";
    for (int i = 1; i <= 10; i++) {
        std::cout << std::setw(4) << i;
    }
    std::cout << std::endl << "    ";
    for (int i = 1; i <= 40; i++) std::cout << "-";
    std::cout << std::endl;

    for (int i = 1; i <= 10; i++) {
        std::cout << std::setw(2) << i << " |";
        for (int j = 1; j <= 10; j++) {
            std::cout << std::setw(4) << (i * j);
        }
        std::cout << std::endl;
    }
    return 0;
}

13.4 Підсумки лекції

  • Вкладені цикли — цикл всередині циклу
  • Зовнішній цикл керує рядками, внутрішній — стовпцями
  • Використання для малювання фігур та матриць
  • Комбінація while та for можлива
