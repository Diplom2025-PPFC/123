52.1 Rvalue та Lvalue

lvalue — це вираз, що має адресу в пам'яті. rvalue — тимчасове значення без адреси.

#include <iostream>
#include <string>
#include <vector>

int main() {
    int x = 5;
    int& lref = x;        // lvalue reference
    // int& badRef = 5;   // ПОМИЛКА!

    int&& rref = 5;       // rvalue reference
    int&& rref2 = x + 10; // OK, тимчасове значення

    // std::move — перетворення lvalue в rvalue
    std::string s1 = "Hello";
    std::string s2 = std::move(s1);  // Переміщення, не копіювання!

    std::cout << "s1: '" << s1 << "'" << std::endl;  // порожній або невизначений
    std::cout << "s2: '" << s2 << "'" << std::endl;  // Hello

    return 0;
}
