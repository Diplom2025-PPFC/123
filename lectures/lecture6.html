Рис. 6.1 — Арифметичні операції в C++

6.1 Базові арифметичні оператори

ОператорОпераціяПрикладРезультат
+Додавання5 + 38
-Віднімання5 - 32
*Множення5 * 315
/Ділення5 / 31 (цілі числа!)
%Остача від ділення5 % 32
#include <iostream>

int main() {
    int a = 17, b = 5;

    std::cout << "a = " << a << ", b = " << b << std::endl;
    std::cout << "a + b = " << (a + b) << std::endl;
    std::cout << "a - b = " << (a - b) << std::endl;
    std::cout << "a * b = " << (a * b) << std::endl;
    std::cout << "a / b = " << (a / b) << std::endl;  // ЦІЛЕ ЧИСЛО: 3
    std::cout << "a % b = " << (a % b) << std::endl;  // Остача: 2

    // Для отримання дробового результату — перетворення типу
    std::cout << "(double)a / b = " << (static_cast<double>(a) / b) << std::endl;
    // Результат: 3.4

    return 0;
}
⚠️ Важливо: цілочисельне ділення

Коли обидва операнди цілі, оператор / виконує цілочисельне ділення (відкидає дробову частину). Щоб отримати дробовий результат, хоча б один операнд має бути дробового типу.

6.2 Оператори інкременту та декременту

ОператорНазваОпис
++xПрефіксний інкрементСпочатку збільшує, потім повертає значення
x++Постфіксний інкрементСпочатку повертає значення, потім збільшує
--xПрефіксний декрементСпочатку зменшує, потім повертає значення
x--Постфіксний декрементСпочатку повертає значення, потім зменшує
#include <iostream>

int main() {
    int a = 5, b = 5;

    // Префіксний інкремент
    std::cout << "++a = " << ++a << std::endl;  // 6 (a стало 6, вивелось 6)
    std::cout << "a = " << a << std::endl;      // 6

    // Постфіксний інкремент
    std::cout << "b++ = " << b++ << std::endl;  // 5 (вивелось 5, потім b стало 6)
    std::cout << "b = " << b << std::endl;      // 6

    return 0;
}

6.3 Скорочені оператори присвоєння

int x = 10;

x += 5;   // x = x + 5;    → 15
x -= 3;   // x = x - 3;    → 12
x *= 2;   // x = x * 2;    → 24
x /= 4;   // x = x / 4;    → 6
x %= 4;   // x = x % 4;    → 2

6.5 Математичні функції

#include <iostream>
#include <cmath>

int main() {
    double x = 16.0, y = 2.0;

    std::cout << "sqrt(16) = " << std::sqrt(x) << std::endl;    // 4
    std::cout << "pow(2, 8) = " << std::pow(2.0, 8.0) << std::endl; // 256
    std::cout << "abs(-10) = " << std::abs(-10) << std::endl;     // 10
    std::cout << "ceil(3.2) = " << std::ceil(3.2) << std::endl;   // 4
    std::cout << "floor(3.9) = " << std::floor(3.9) << std::endl; // 3
    std::cout << "round(3.5) = " << std::round(3.5) << std::endl; // 4
    std::cout << "sin(PI/2) = " << std::sin(3.14159/2) << std::endl; // ~1
    std::cout << "log(100) = " << std::log(100.0) << std::endl;    // натуральний
    std::cout << "log10(100) = " << std::log10(100.0) << std::endl; // 2

    return 0;
}
