60.1 Патерн Singleton

Singleton гарантує, що клас має лише один екземпляр і надає глобальний доступ до нього.

#include <iostream>
#include <string>

class Logger {
private:
    static Logger* instance;
    Logger() {}  // Приватний конструктор

public:
    static Logger* getInstance() {
        if (instance == nullptr) {
            instance = new Logger();
        }
        return instance;
    }

    void log(const std::string& message) {
        std::cout << "[LOG] " << message << std::endl;
    }

    // Заборонити копіювання
    Logger(const Logger&) = delete;
    Logger& operator=(const Logger&) = delete;
};

Logger* Logger::instance = nullptr;

// Патерн Factory
class Animal {
public:
    virtual void speak() = 0;
    virtual ~Animal() {}
};

class Dog : public Animal {
public:
    void speak() override { std::cout << "Гав!" << std::endl; }
};

class Cat : public Animal {
public:
    void speak() override { std::cout << "Мяу!" << std::endl; }
};

class AnimalFactory {
public:
    static Animal* create(const std::string& type) {
        if (type == "dog") return new Dog();
        if (type == "cat") return new Cat();
        return nullptr;
    }
};

int main() {
    // Singleton
    Logger::getInstance()->log("Програма старт");

    // Factory
    Animal* dog = AnimalFactory::create("dog");
    Animal* cat = AnimalFactory::create("cat");

    dog->speak();  // Гав!
    cat->speak();  // Мяу!

    delete dog;
    delete cat;

    return 0;
}

60.2 Підсумки лекції

  • Singleton — один екземпляр
  • Factory — створення об'єктів
  • Observer — підписка на події
  • Strategy — зміна алгоритмів
